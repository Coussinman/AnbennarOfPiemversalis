country_event = {
    id = aw_minotaurs.1
    title = aw_minotaurs.1.t
    desc = aw_minotaurs.1.d
    picture = COMET_SIGHTED_eventPicture
    
    hidden = yes
    is_triggered_only = yes
    
    trigger = {
        any_owned_province = {
            OR = { 
                has_province_modifier = aw_minotaurs_1
                has_province_modifier = aw_minotaurs_2
                has_province_modifier = aw_minotaurs_3
            }
        }
    }

    mean_time_to_happen = {
        days = 1
        modifier = {
            factor = 2
            num_of_owned_provinces_with = {
                value = 3

                    has_province_modifier = aw_minotaurs_1
                    has_province_modifier = aw_minotaurs_2
                    has_province_modifier = aw_minotaurs_3
                }
            }
        }
    }

    immediate = {
        random_owned_province = {
            limit = {
                OR = { 
                    has_province_modifier = aw_minotaurs_1
                    has_province_modifier = aw_minotaurs_2
                    has_province_modifier = aw_minotaurs_3
                }
            }
            save_event_target_as = aw_minotaurs_target
        }
    }
    
	
	option = {
        event_target:aw_minotaurs_target = {
            random_list = {

                7 = {
                    province_event = { id = aw_minotaurs.20 }    #Natural removal

                #21 - adventurer removal
                15 = {
                    #Base modifiers

                        factor = 2
                        OR = {
                            owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }
                            owner = { has_adventurer_reform = yes }
                            owner = { has_estate_privilege = estate_ahati_support_south }
							owner = { has_estate_privilege = estate_ahati_support_brotherhoods }
                        }
                    }

                    #more likely if loyal
                    modifier = {
                        factor = 2
                        owner = {
                            estate_loyalty = {
                                estate = estate_adventurers
                                loyalty = 60
                            }
                        }
                    }

                    #more likely if high influence
                    modifier = {
                        factor = 2
                        owner = {
                            estate_influence = {
                                estate = estate_adventurers
                                influence = 60
                            }
                        }
                    }
                    #small influence boost at 30
                    modifier = {
                        factor = 1.5
                        owner = {
                            estate_influence = {
                                estate = estate_adventurers
                                influence = 30
                            }
                        }
                    }
                    #adventurer type sponsorships
                    modifier = {
                        factor = 2
                        owner = {
                            has_estate_privilege = estate_adventurers_sponsor_beast_slayers
                        }
                    }
                    province_event = { id = aw_minotaurs.21 }
                }



                #24 - monstrous nation removal
                50 = {
                    modifier = {
                        #only want this to ever be part of the equation if you are monstrous
                        factor = 0
                        NOT = { owner = { has_country_modifier = monstrous_nation } }
                    }
                    #Base modifiers
                    province_event = { id = aw_minotaurs.24 }
                }

                #120 - Adventurer Peaceful Resolution
                10 = {
                    #more likely if generous quest rewards/adverturer nation
                    modifier = {
                        factor = 2
                        OR = {
                            owner = { has_estate_privilege = estate_adventurers_generous_quest_rewards }

    goto = ROOT
    
    is_triggered_only = yes
    
    trigger = {
        owner = { OR = { has_estate = estate_adventurers has_adventurer_reform = yes  } }
        OR = {
            has_province_modifier = aw_minotaurs_1
            has_province_modifier = aw_minotaurs_2
            has_province_modifier = aw_minotaurs_3
        }
    }
    
    option = {
        # oh, thank goodness
        name = aw_minotaurs.21.a
        ai_chance = { factor = 100 }

        remove_province_modifier = aw_minotaurs_1
        remove_province_modifier = aw_minotaurs_2
        remove_province_modifier = aw_minotaurs_3

        owner = {
            if = {
                limit = { has_estate = estate_adventurers }
                add_estate_influence_modifier = {
                    estate = estate_adventurers
                    desc = EST_VAL_AW_RESOLVED_ADVENTURERS_WANTED_ENCOUNTER
                    duration = 3650
                    influence = 5
                }
            }
        }

        owner = { add_prestige_excess_to_splendour_effect = { VAL=3} }

    }
}

province_event = {
    id = aw_minotaurs.122
    title = aw_minotaurs.122.t
    desc = aw_minotaurs.122.d
    picture = ANGRY_MOB_eventPicture
    goto = ROOT
    
    is_triggered_only = yes
    
    trigger = {
        OR = {
            has_province_modifier = aw_minotaurs_1
            has_province_modifier = aw_minotaurs_2
        }

    }
    
    option = {
        name = aw_minotaurs.122.a
        ai_chance = { factor = 100 }

        if = {
            limit = {
                has_province_modifier = aw_minotaurs_1
            }
            remove_province_modifier = aw_minotaurs_1
            add_province_modifier = {
                name = aw_minotaurs_2
                duration = -1
            }
        }
        if = {
            limit = {
                has_province_modifier = aw_minotaurs_2
            }
            remove_province_modifier = aw_minotaurs_2
            add_province_modifier = {
                name = aw_minotaurs_3
                duration = -1
            }
        }
    }

    option = {
        name = aw_minotaurs.122.b
        ai_chance = { factor = 100 }
        owner = { add_manpower = -0.05 }
        random_list = {
            50 = {
                custom_tooltip = ACE_minotaurs_the_attack_succeeds_tt
                owner = { add_army_tradition_or_mil_power = yes }
            }
            50 = {
                custom_tooltip = ACE_minotaurs_the_attack_fails_tt
                if = {
                    limit = {
                        has_province_modifier = aw_minotaurs_1
                    }
                    remove_province_modifier = aw_minotaurs_1
                    add_province_modifier = {
                        name = aw_minotaurs_2
                        duration = -1
                    }
                }
                else_if = {
                    limit = {
                        has_province_modifier = aw_minotaurs_2
                    }
                    remove_province_modifier = aw_minotaurs_2
                    add_province_modifier = {
                        name = aw_minotaurs_3
                        duration = -1
                    }
                }
            }
        }
    }

    option = {
        name = aw_minotaurs.122.c
        ai_chance = { factor = 100 }
        highlight = yes
        trigger = {
            owner = {
                has_country_modifier = ACE_minotaur_tamers
            }
        }
        remove_province_modifier = aw_minotaurs_1
        remove_province_modifier = aw_minotaurs_2
        remove_province_modifier = aw_minotaurs_3

        add_province_modifier = {
            name = aw_tamed_minotaur_herd
            duration = 3650
        }
    }
