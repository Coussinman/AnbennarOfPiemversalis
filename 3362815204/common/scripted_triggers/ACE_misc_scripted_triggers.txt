	}

warless_victory_trigger = {
	has_dlc = "Art of War"
	is_emperor = yes
	is_at_war = no
	NOT = {
		any_known_country = {
			is_in_league_war = yes
		}
	}
	has_regency = no
	hre_religion_locked = no
	religion = $religion$
	OR = {
		AND = {
			hre_leagues_enabled = yes
			had_global_flag = {
				flag = evangelical_union_happened
				days = 10950
			}
			NOT = {
				any_known_country = {
					is_elector = yes
					OR = {
						[[regent_court]
							religion = corinite
							religion = ravelian
							religion = dalcabba
						]
						[[corinite]
							religion = regent_court
							religion = ravelian
							religion = dalcabba
						]
						[[ravelian]
							religion = regent_court
							religion = corinite
							religion = dalcabba
						]
						[[dalcabba]
							religion = regent_court
							religion = corinite
							religion = ravelian
						]
					}
					is_league_enemy = ROOT
					truce_with = ROOT
				}
			}
		}
		AND = {
			NOT = {
				has_global_flag = evangelical_union_happened
			}
			is_year = 1610
		}
	}
}
