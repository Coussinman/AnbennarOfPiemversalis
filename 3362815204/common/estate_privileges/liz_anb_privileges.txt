
#Mages

estate_mages_steppe_enhancement = {
	icon = privilege_promote_maratha_nobility
	max_absolutism = -5
	conditional_modifier = { trigger = { has_absolutism_reduction_for_estate_privileges = yes } modifier = { max_absolutism = 2 } }
	influence = 0.05
	loyalty = 0.05
	land_share = 0

	is_valid = {
    	has_country_flag = zirbit_2
	}

	trigger = {
        has_country_flag = zirbit_2
	}
	
	can_revoke = {
		always = yes
	}

	penalties = {
	}

	benefits = {
	}

	mechanics = {
	}
	
	on_granted = {
		custom_tooltip = estate_mages_steppe_enhancement_tooltip
	}

	on_granted_province = {
		if = {
			limit = { has_terrain = steppe }
				add_province_modifier = {
					name = mages_steppe_dev_cost
					duration = -1
				}
		}
	}

	on_revoked = {
		custom_tooltip = estate_mages_steppe_enhancement_revoke_tooltip
	}

	on_revoked_province = {
		if = {
			limit = { has_terrain = steppe }
			remove_province_modifier = mages_steppe_dev_cost
		}
	}
	
	ai_will_do = { factor = 0 }
}
