cb_glp_faction_war = {
	valid_for_subject = no

	prerequisites_self = {
    	has_global_flag = glp_factions_event_started
		ai = no
		OR = {
			has_country_flag = glp_factions_joined_sideA_flag
			has_country_flag = glp_factions_joined_sideB_flag
		}
	}

	prerequisites = {
		FROM = { ai = no }
		if = {
			limit = {
				has_country_flag = glp_factions_joined_sideA_flag
			}
			FROM = {
				has_country_flag = glp_factions_joined_sideB_flag
			}
		}
		else = {
			limit = {
				has_country_flag = glp_factions_joined_sideB_flag
			}
			FROM = {
				has_country_flag = glp_factions_joined_sideA_flag
			}
		}
	}

	war_goal = glp_faction_war_wargoal
}
