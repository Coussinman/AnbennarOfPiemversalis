ACE_game_rules_toggle_region_effect = {
	hidden_effect = {
		if = {
			limit = {
				is_variable_equal = {
					which = $var$
					value = 3
				}
			}
			set_variable = {
				which = $var$
				value = 1
			}
		}
		else = {
			change_variable = {
				which = $var$
				value = 1
			}
		}
	}
}

ACE_game_rules_back_option_effect = {
	hidden_effect = {
		if = {
			limit = {
				is_variable_equal = {
					which = $var$
					value = 1
				}
			}
			set_variable = {
				which = $var$
				value = $num$
			}
		}
		else = {
			subtract_variable = {
				which = $var$
				value = 1
			}
		}
	}
