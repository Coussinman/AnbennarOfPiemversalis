B94_increase_xamarzu_blessing_effect = {
	if = {
		limit = {
			NOT = {
				OR = {
					has_ruler_modifier = B94_ruler_blessing_1
					has_ruler_modifier = B94_ruler_blessing_2
					has_ruler_modifier = B94_ruler_blessing_3
					has_ruler_modifier = B94_ruler_blessing_4
					has_ruler_modifier = B94_ruler_blessing_5
				}
			}
		}
		add_ruler_modifier = {
			name = B94_ruler_blessing_1
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_ruler_modifier = B94_ruler_blessing_1
		}
		remove_country_modifier = B94_ruler_blessing_1
		add_ruler_modifier = {
			name = B94_ruler_blessing_2
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_ruler_modifier = B94_ruler_blessing_2
		}
		remove_country_modifier = B94_ruler_blessing_2
		add_ruler_modifier = {
			name = B94_ruler_blessing_3
			duration = -1
		}
	}
	else_if = {
		limit = {
			has_ruler_modifier = B94_ruler_blessing_3
		}
		remove_country_modifier = B94_ruler_blessing_3
		add_ruler_modifier = {
			name = B94_ruler_blessing_4
			duration = -1
		}
	}
	else = {
		remove_country_modifier = B94_ruler_blessing_4
		add_ruler_modifier = {
			name = B94_ruler_blessing_5
			duration = -1
		}
	}
}
