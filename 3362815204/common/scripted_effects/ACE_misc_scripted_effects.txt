religious_supremacy_provincial_modifier = {
	custom_tooltip = ACE_remove_religious_zeal_tooltip
	hidden_effect = {
		every_province = {
			limit = {
				owner = {
					is_part_of_hre = yes
				}
				is_part_of_hre = yes
				religion = $religion$
			}
			add_province_modifier = {
				name = religious_zeal_at_conv
				duration = 10950
			}
		}
		every_province = {
			limit = {
				owner = {
					religion = $religion$
					is_part_of_hre = yes
				}
				is_part_of_hre = yes
				NOT {
					religion = $religion$
				}
			}
			remove_province_modifier = religious_zeal_at_conv
		}
	}
	custom_tooltip = ace_remove_centers_tooltip
	add_country_modifier = {
		name = ace_removed_centers
		duration = 10950
	}
	europe = {
		limit = {
			is_part_of_hre = yes
		}
		[[regent_court]
			remove_reform_center = corinite
			remove_reform_center = ravelian
			remove_reform_center = dalcabba
		]
		[[corinite]
			remove_reform_center = regent_court
			remove_reform_center = ravelian
			remove_reform_center = dalcabba
		]
		[[ravelian]
			remove_reform_center = regent_court
			remove_reform_center = corinite
			remove_reform_center = dalcabba
		]
		[[dalcabba]
			remove_reform_center = regent_court
			remove_reform_center = corinite
			remove_reform_center = ravelian
		]
	}
}
