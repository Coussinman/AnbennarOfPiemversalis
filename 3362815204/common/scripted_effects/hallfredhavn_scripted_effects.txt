
hallfredhavn_expand_slavery = {
	custom_tooltip = AD5_expand_slavery_tt
	hidden_effect = {
		if = {
			limit = {
				has_province_modifier = AD5_huge_langeyru_slavery
			}
			add_base_production = 1
		}
		else_if = {
			limit = {
				has_province_modifier = AD5_significant_langeyru_slavery
			}
			remove_province_modifier = AD5_significant_langeyru_slavery
			add_province_modifier = {
				name = AD5_huge_langeyru_slavery
				duration = -1
			}
		}
		else_if = {
			limit = {
				has_province_modifier = AD5_small_langeyru_slavery
			}
			remove_province_modifier = AD5_small_langeyru_slavery
			add_province_modifier = {
				name = AD5_significant_langeyru_slavery
				duration = -1
			}
		}
		else = {
			add_province_modifier = {
				name = AD5_small_langeyru_slavery
				duration = -1
			}
		}
	}
}
