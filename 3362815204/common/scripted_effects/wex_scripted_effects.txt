alenize_entire_culture = {
	#move cultures to alenic
	if = {
		limit = {
			has_consort = yes
			consort_culture = $culture$
		}
		set_consort_culture = $culture$_alenic
	}
	hidden_effect = {
		every_province = {
			limit = {
				culture = $culture$
			}
			change_culture = $culture$_alenic
		}
		every_country = {
			limit = {
				primary_culture = $culture$
			}
			change_primary_culture = $culture$_alenic
			set_ruler_culture = $culture$_alenic
			if = {
				limit = {
					has_heir = yes
				}
				set_heir_culture = $culture$_alenic
			}
			if = {
				limit = {
					has_consort = yes
					consort_culture = $culture$
				}
				set_consort_culture = $culture$_alenic
			}
		}
		set_global_flag = alenicised_$culture$_flag
		[[clear_previous_accepted_culture]
		remove_accepted_culture = $culture$
		]
	}
}
