#Gnomish_resettlers for all 16 provinces it checks if it is owned by reveria, if not, it skips the province, if it is owned by reveria it adds -1 dev to 169 and then checks if the culture is reverian, if so, it adds a gnomish minority, if not, it makes the province gnomish
country_event = {
	id = reveria.14
	title = reveria.14.t
	desc = reveria.14.d
	picture = VILLAGE_BURNING_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = reveria.14.a
		ai_chance = { 
			factor = 100 
		}
		if = { 
			limit = { owns_core_province = 167 }
			if = {
				limit = { 167 = { culture = ROOT } }
				167 = { add_gnomish_minority_size_effect = yes } 
			}
			else = { 
				167 = { 
					change_culture = cliff_gnome 
					change_religion = the_thought 
					add_human_minority_size_effect = yes
				}
			}
			169 = { add_base_manpower = -1 }
			167 = { add_base_manpower = 1 }		
		}
		if = { 
			limit = { owns_core_province = 170 }
			if = {
				limit = { 170 = { culture = ROOT } }
				170 = { add_gnomish_minority_size_effect = yes } 		
			}
			else = { 
				170 = {
					change_culture = cliff_gnome 
					change_religion = the_thought 
					add_human_minority_size_effect = yes
